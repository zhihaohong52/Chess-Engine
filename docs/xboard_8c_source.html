61int checkresult(S_BOARD *pos) {

63 if (pos->fiftyMove > 100) {

74 printf("1/2-1/2 {insufficient material (claimed by Vice)}\n");

78 S_MOVELIST list[1];

82 int found = 0;
83 for (MoveNum = 0; MoveNum < list->count; ++MoveNum) {

85 if (!MakeMove(pos, list->moves[MoveNum].move)) {
86 continue;
87 }
88 found++;
89 TakeMove(pos);
90 break;
91 }
92
93 if (found != 0) return FALSE;
94
95 int InCheck = SqAttacked(pos->KingSq[pos->side], pos->side^1, pos);
96
97 if (InCheck == TRUE) {
98 if (pos->side == WHITE) {
99 printf("0-1 {black mates (claimed by Vice)}\n");
100 return TRUE;
101 } else {
102 printf("0-1 {white mates (claimed by Vice)}\n");
103 return TRUE;
104 }
105 } else {
106 printf("\n1/2-1/2 {stalemate (claimed by Vice)}\n");
107 return TRUE;
108 }
109
110 return FALSE;
111}
